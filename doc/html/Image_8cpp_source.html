167 void Image::afficher()
168 {
169  SDL_Event windowEvent;
170  bool quit = false;
171 
172  while (!quit)
173  {
174  if (SDL_PollEvent(&windowEvent))
175  {
176  if (windowEvent.type == SDL_QUIT)
177  {
178  quit = true;
179  }
180  if(windowEvent.type == SDL_KEYDOWN)
181  {
182  switch(windowEvent.key.keysym.scancode)
183  {
184  case SDL_SCANCODE_ESCAPE:
185  if(texture != nullptr) SDL_DestroyTexture(texture);
186  if(renderer != nullptr) SDL_DestroyRenderer(renderer);
187  if(window != nullptr) SDL_DestroyWindow(window);
188  SDL_Quit();
189  quit = true;
190  break;
191 
192  case SDL_SCANCODE_T:
193  zoomLevel++;
194  if(zoomLevel > 19) zoomLevel = 19;
195  break;
196 
197  case SDL_SCANCODE_G:
198  zoomLevel--;
199  if(zoomLevel < 0) zoomLevel = 0;
200  break;
201 
202  default:
203  break;
204  }
205  }
206  }
207 
208  if(quit) break;
209 
210  int col = 169;
211  SDL_SetRenderDrawColor(renderer, col, col, col, 255);
212  SDL_RenderClear(renderer);
213 
214  if(SDL_SetRenderTarget(renderer, texture) != 0)
215  {
216  cout<<"Erreur SDL_SetRenderTarget : "<< SDL_GetError() <<endl;
217  exit(1);
218  }
219  for (unsigned int i = 0; i < dimx; ++i)
220  {
221  for (unsigned int j = 0; j < dimy; ++j)
222  {
223  const Pixel p = tab[dimx * j + i];
224  SDL_SetRenderDrawColor(renderer, p.getRouge(), p.getVert(), p.getBleu(), 255);
225  SDL_RenderDrawPoint(renderer, i, j);
226  }
227  }
228  SDL_SetRenderTarget(renderer, NULL);
229 
230  SDL_Rect r;
231  r.x = 0;
232  r.y = 0;
233  r.w = dimx + zoomIncrement * zoomLevel;
234  r.h = dimy + zoomIncrement * zoomLevel;
235 
236  if(r.w < (signed int) dimx) r.w = dimx;
237  if(r.h < (signed int) dimy) r.h = dimy;
238  if(r.w > WIDTH) r.w = WIDTH;
239  if(r.h > HEIGHT) r.h = HEIGHT;
240 
241  r.x = (r.x-r.w/2) + WIDTH/2;
242  r.y = (r.y-r.h/2) + HEIGHT/2;
243  SDL_RenderCopy(renderer, texture, NULL, &r);
244 
245  SDL_RenderPresent(renderer);
246  }
